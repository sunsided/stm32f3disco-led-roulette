    \node[anchor=south west,inner sep=0] (image) at (0,0) {\includegraphics[width=10cm]{\imagepath}};

    % Get the width and height of the image
    \begin{scope}

        \pgfextractx{\imagewidth}{\pgfpointanchor{image}{north east}}
        \pgfextracty{\imageheight}{\pgfpointanchor{image}{north east}}
    \end{scope}

    % Define 3D coordinate system
    \begin{scope}[x={(image.north east)}, y={(image.south west)}, z={(0cm,0cm,1cm)}, canvas is xy plane at z=0]
